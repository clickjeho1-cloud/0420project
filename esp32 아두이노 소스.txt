#include <WiFi.h>
#include <PubSubClient.h>
#include <ArduinoJson.h>
#include <DHT.h>

// ===== WiFi =====
const char* ssid = "daesin_302";
const char* password = "ds123456";

// ===== MQTT =====
const char* mqtt_server = "763d603e502d4671a5c950470203ec7f.s1.eu.hivemq.cloud";
const int mqtt_port = 1883;
const char* mqtt_user = "jhk001";
const char* mqtt_pass = "Sinwonpark1!";

// ===== Topic =====
const char* topic_pub = "smartfarm/jeho123/data";
const char* topic_sub = "smartfarm/jeho123/control";

// ===== DHT =====
#define DHTPIN 2
#define DHTTYPE DHT22
DHT dht(DHTPIN, DHTTYPE);

// ===== EC / pH (아날로그) =====
#define EC_PIN 34
#define PH_PIN 35

// ===== 핀 =====
#define FAN_PIN 4
#define PUMP_PIN 5
#define LED_PIN 6
#define HEATER_PIN 7

// ===== MQTT =====
WiFiClient espClient;
PubSubClient client(espClient);

// ===== 스케줄 =====
struct Schedule {
  int startH;
  int startM;
  int endH;
  int endM;
  bool active;
};

Schedule pumpSchedule = {8, 0, 9, 0, true};

// ===== 장치 제어 =====
void applyDevices(JsonObject devices) {

  if (devices.containsKey("fan")) {
    int pwm = devices["fan"]["pwm"];
    ledcWrite(0, map(pwm, 0, 100, 0, 255));
  }

  if (devices.containsKey("pump")) {
    bool on = devices["pump"]["on"];
    digitalWrite(PUMP_PIN, on ? HIGH : LOW);
  }

  if (devices.containsKey("led")) {
    int pwm = devices["led"]["pwm"];
    ledcWrite(1, map(pwm, 0, 100, 0, 255));
  }

  if (devices.containsKey("heater")) {
    bool on = devices["heater"]["on"];
    digitalWrite(HEATER_PIN, on);
  }
}

// ===== MQTT =====
void callback(char* topic, byte* payload, unsigned int length) {

  StaticJsonDocument<512> doc;
  deserializeJson(doc, payload, length);

  String cmd = doc["cmd_type"];

  if (cmd == "manual" || cmd == "auto") {
    applyDevices(doc["devices"]);
  }

  // ===== schedule 등록 =====
  if (cmd == "schedule") {
    pumpSchedule.startH = doc["start_hour"];
    pumpSchedule.startM = doc["start_min"];
    pumpSchedule.endH = doc["end_hour"];
    pumpSchedule.endM = doc["end_min"];
  }
}

// ===== WiFi + MQTT =====
void setup() {
  Serial.begin(115200);

  pinMode(PUMP_PIN, OUTPUT);
  pinMode(HEATER_PIN, OUTPUT);

  ledcSetup(0, 5000, 8);
  ledcAttachPin(FAN_PIN, 0);

  ledcSetup(1, 5000, 8);
  ledcAttachPin(LED_PIN, 1);

  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) delay(500);

  client.setServer(mqtt_server, mqtt_port);
  client.setCallback(callback);

  dht.begin();
}

void reconnect() {
  while (!client.connected()) {
    client.connect("esp32", mqtt_user, mqtt_pass);
    client.subscribe(topic_sub);
  }
}

// ===== LOOP =====
void loop() {

  if (!client.connected()) reconnect();
  client.loop();

  // ===== 센서 =====
  float temp = dht.readTemperature();
  float hum = dht.readHumidity();

  int ec_raw = analogRead(EC_PIN);
  int ph_raw = analogRead(PH_PIN);

  float ec = ec_raw * 0.01;
  float ph = ph_raw * 0.01;

  // ===== JSON 송신 =====
  StaticJsonDocument<256> doc;

  doc["temperature"] = temp;
  doc["humidity"] = hum;
  doc["ec"] = ec;
  doc["ph"] = ph;

  char buf[256];
  serializeJson(doc, buf);

  client.publish(topic_pub, buf);

  delay(2000);
}
