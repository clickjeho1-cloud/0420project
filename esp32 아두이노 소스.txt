#include <WiFi.h>
#include <PubSubClient.h>
#include <ArduinoJson.h>
#include <DHT.h>

// ===== WiFi =====
const char* ssid = "daesin_302";
const char* password = "ds123456";

// ===== MQTT =====
const char* mqtt_server = "763d603e502d4671a5c950470203ec7f.s1.eu.hivemq.cloud";
const int mqtt_port = 1883;
const char* mqtt_user = "jhk001";
const char* mqtt_pass = "Sinwonpark1!";

// ===== Topic =====
const char* topic_pub = "smartfarm/jeho123/data";
const char* topic_sub = "smartfarm/jeho123/control";

// ===== DHT =====
#define DHTPIN 2
#define DHTTYPE DHT22
DHT dht(DHTPIN, DHTTYPE);

// ===== PID =====
float setpoint = 25.0;
float Kp = 2.5;
float Ki = 0.6;
float Kd = 1.2;

float e_prev = 0;
float integral = 0;

// ===== 핀 =====
const int fanPin = 4;
const int pumpPin = 5;
const int ledPin = 6;
const int heaterPin = 7;

// ===== PWM 채널 =====
#define FAN_CH 0
#define LED_CH 1
#define HEATER_CH 2

// ===== 펌프 논블로킹 =====
bool pumpRunning = false;
unsigned long pumpStart = 0;
unsigned long pumpDuration = 0;

// ===== MQTT =====
WiFiClient espClient;
PubSubClient client(espClient);

// ===== 장치 제어 =====
void handleDevices(JsonObject devices) {

  // FAN
  if (devices.containsKey("fan")) {
    int pwm = devices["fan"]["pwm"] | 0;
    ledcWrite(FAN_CH, map(pwm, 0, 100, 0, 255));
  }

  // PUMP
  if (devices.containsKey("pump")) {
    bool on = devices["pump"]["on"] | false;
    int duration = devices["pump"]["duration_sec"] | 0;

    if (on) {
      digitalWrite(pumpPin, HIGH);
      pumpDuration = duration * 1000;
      pumpStart = millis();
      pumpRunning = true;
    }
  }

  // LED
  if (devices.containsKey("led")) {
    int pwm = devices["led"]["pwm"] | 0;
    ledcWrite(LED_CH, map(pwm, 0, 100, 0, 255));
  }

  // HEATER
  if (devices.containsKey("heater")) {
    bool on = devices["heater"]["on"] | false;
    ledcWrite(HEATER_CH, on ? 255 : 0);
  }
}

// ===== MQTT 수신 =====
void callback(char* topic, byte* payload, unsigned int length) {

  StaticJsonDocument<512> doc;
  DeserializationError err = deserializeJson(doc, payload, length);

  if (err) {
    Serial.println("JSON parse error");
    return;
  }

  const char* cmd = doc["cmd_type"];

  if (cmd && strcmp(cmd, "manual") == 0) {
    JsonObject devices = doc["devices"];
    handleDevices(devices);
    Serial.println("CMD APPLIED");
  }
}

// ===== MQTT 재연결 =====
void reconnect() {
  while (!client.connected()) {
    Serial.print("MQTT connecting...");

    if (client.connect("esp32_client", mqtt_user, mqtt_pass)) {
      Serial.println("OK");
      client.subscribe(topic_sub);
    } else {
      Serial.print("FAIL: ");
      Serial.println(client.state());
      delay(2000);
    }
  }
}

// ===== SETUP =====
void setup() {
  Serial.begin(115200);

  pinMode(pumpPin, OUTPUT);

  // PWM 설정
  ledcSetup(FAN_CH, 5000, 8);
  ledcAttachPin(fanPin, FAN_CH);

  ledcSetup(LED_CH, 5000, 8);
  ledcAttachPin(ledPin, LED_CH);

  ledcSetup(HEATER_CH, 5000, 8);
  ledcAttachPin(heaterPin, HEATER_CH);

  // WiFi 연결
  WiFi.begin(ssid, password);

  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  Serial.println("\nWiFi connected");

  client.setServer(mqtt_server, mqtt_port);
  client.setCallback(callback);

  dht.begin();
}

// ===== LOOP =====
void loop() {

  if (!client.connected()) reconnect();
  client.loop();

  // ===== 펌프 종료 =====
  if (pumpRunning && millis() - pumpStart > pumpDuration) {
    digitalWrite(pumpPin, LOW);
    pumpRunning = false;
  }

  static unsigned long lastTime = 0;

  if (millis() - lastTime > 1000) {
    lastTime = millis();

    float temp = dht.readTemperature();
    if (isnan(temp)) return;

    // ===== PID =====
    float error = setpoint - temp;
    integral += error;

    // anti-windup
    if (integral > 100) integral = 100;
    if (integral < -100) integral = -100;

    float derivative = error - e_prev;

    float output = Kp * error + Ki * integral + Kd * derivative;

    if (output > 255) output = 255;
    if (output < 0) output = 0;

    e_prev = error;

    // ===== JSON 송신 =====
    StaticJsonDocument<256> doc;
    doc["time"] = millis();

    JsonObject v = doc.createNestedObject("values");
    v["temperature"] = temp;
    v["error"] = error;
    v["integral"] = integral;
    v["derivative"] = derivative;
    v["output"] = output;

    char buffer[256];
    serializeJson(doc, buffer);

    client.publish(topic_pub, buffer);

    Serial.println(buffer);

    // ===== 히터 제어 =====
    ledcWrite(HEATER_CH, (int)output);
  }
}
