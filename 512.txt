// =====================================================
// CRC 자동 계산
// =====================================================
uint16_t calcCRC(uint8_t* buf, int len)
{
  uint16_t crc = 0xFFFF;
  for (int i = 0; i < len; i++)
  {
    crc ^= buf[i];
    for (int j = 0; j < 8; j++)
    {
      if (crc & 0x0001)
        crc = (crc >> 1) ^ 0xA001;
      else
        crc >>= 1;
    }
  }
  return crc;
}

// =====================================================
// READ EC (에코 무시 및 수동 DE/RE 제어 적용)
// =====================================================
float readEC()
{
  // BUFFER CLEAR
  while (RS485Serial.available()) {
    RS485Serial.read();
  }

  // MODBUS REQUEST 프레임 (주소 0x0001, 필요시 0x0015나 0x0002로 변경 테스트)
  uint8_t frame[6] = {
    0x01,  // Device ID
    0x03,  // Read Holding Registers
    0x00, 0x01,  // 레지스터 주소 0x0001 (EC값)
    0x00, 0x01   // 1개 읽기
  };

  // CRC 자동 계산 및 결합
  uint16_t crc = calcCRC(frame, 6);
  uint8_t request[8];
  memcpy(request, frame, 6);
  request[6] = crc & 0xFF;         // CRC Low
  request[7] = (crc >> 8) & 0xFF;  // CRC High

  // DEBUG 송신 프레임 출력
  Serial.print("EC REQUEST  : ");
  for (int i = 0; i < 8; i++) {
    Serial.print("0x");
    if (request[i] < 16) Serial.print("0");
    Serial.print(request[i], HEX);
    Serial.print(" ");
  }
  Serial.println();

  // 💡 [핵심] MAX485 수동 제어 (송신 모드)
  digitalWrite(RS485_CONTROL_PIN, HIGH); // 송신 모드로 핀을 HIGH
  delay(2); // 모드가 완전히 전환될 때까지 짧은 대기
  
  // SEND
  RS485Serial.write(request, 8);
  RS485Serial.flush(); // 데이터가 선을 타고 다 나갈 때까지 대기

  // 💡 [핵심] MAX485 수동 제어 (수신 모드 복귀)
  digitalWrite(RS485_CONTROL_PIN, LOW); // 전송이 끝나자마자 수신 모드로 복귀
  delay(2);

  // RECEIVE 루프
  uint8_t response[64]; // 모듈 에코나 노이즈까지 포함해서 넉넉하게 저장
  int idx = 0;
  unsigned long startTime = millis();

  // 넉넉하게 500ms 동안 수신 버퍼로 들어오는 모든 데이터를 모음
  while (millis() - startTime < 500) {
    if (RS485Serial.available()) {
      response[idx++] = RS485Serial.read();
      if (idx >= 64) break; 
    }
  }

  // HEX DEBUG 수신 출력
  Serial.print("EC RAW RESPONSE : ");
  for (int i = 0; i < idx; i++) {
    Serial.print("0x");
    if (response[i] < 16) Serial.print("0");
    Serial.print(response[i], HEX);
    Serial.print(" ");
  }
  Serial.println();

  // 💡 수신된 전체 데이터 중 "01 03 02"로 시작하는 진짜 응답 찾기
  int realDataStart = -1;
  for (int i = 0; i <= idx - 5; i++) { // 최소 5바이트 (헤더3 + 데이터2) 필요
    if (response[i] == 0x01 && response[i+1] == 0x03 && response[i+2] == 0x02) {
      realDataStart = i;
      break;
    }
  }

  // 파싱 (진짜 응답을 찾았을 때만 계산)
  if (realDataStart != -1) {
    int value = (response[realDataStart + 3] << 8) | response[realDataStart + 4];
    float ec  = value / 100.0;

    Serial.print("EC REAL VALUE : ");
    Serial.print(ec);
    Serial.println(" us/cm");

    return ec;
  } else {
    Serial.println("EC ERROR : NO VALID SENSOR RESPONSE FOUND");
    return -1;
  }
}

// =====================================================
// SETUP
// =====================================================
void setup()
{
  Serial.begin(9600); // 시리얼 모니터 통신 속도
  delay(1000);
  Serial.println("\nBOOT OK");

  // 💡 MAX485 제어 핀 초기화
  pinMode(RS485_CONTROL_PIN, OUTPUT);
  digitalWrite(RS485_CONTROL_PIN, LOW); // 기본 상태는 수신(RX) 대기 모드

  dht.begin();

  // RS485 센서 통신 속도 4800으로 설정 (필요시 9600으로 변경)
  RS485Serial.begin(4800, SERIAL_8N1, RX2, TX2);

  Serial.println("WIFI CONNECTING...");
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  Serial.println("\nWIFI CONNECTED");
  Serial.print("IP : ");
  Serial.println(WiFi.localIP());

  espClient.setInsecure();
  client.setServer(mqtt_server, mqtt_port);
}

// =====================================================
// LOOP
// =====================================================
void loop()
{
  if (!client.connected()) {
    reconnect();
  }
  client.loop();

  static unsigned long lastTime = 0;
  if (millis() - lastTime > 2000) {
    lastTime = millis();

    float temp = dht.readTemperature();
    float hum  = dht.readHumidity();

    if (isnan(temp) || isnan(hum)) {
      Serial.println("DHT ERROR");
      return;
    }

    float ecValue = readEC();

    StaticJsonDocument<256> doc;
    doc["temperature"] = temp;
    doc["humidity"]    = hum;
    doc["ec"]          = ecValue;

    char buffer[256];
    serializeJson(doc, buffer);

    bool ok = client.publish(topic_pub, buffer);

    if (ok) Serial.print("SEND OK : ");
    else    Serial.print("SEND FAIL : ");

    Serial.println(buffer);
  }
}
